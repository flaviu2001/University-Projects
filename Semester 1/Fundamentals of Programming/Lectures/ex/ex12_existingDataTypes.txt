'''
    We define two list objects!
        - list is a predefined data type in Python
        - list is implemented as a class!
'''
list_one = [1,2,3]
list_two = [4,5,6]

'''
    The 'append()' method is called on the 'list_one' object
        - list has a __init__ method
        - list has an 'append' method (and many more!)
'''
list_one.append(5)
print(list_one)
print(list_two)
