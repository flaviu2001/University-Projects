revert(l1, l2.. ln, C) = {
	C, n=0
	revert(l2.. ln, [l1]+C), else
}
sublists_aux(l1, l2.. ln, E, C) = {
	C, n=0
	sublists_aux(l2, l3.. ln, H, C+[H]), E < H
	sublists_aux(l2, l3.. ln, E, C) (not else)
}
sublists(l1, l2.. ln) = {
	[], n=0
	sublists_aux(l2, l3.. ln, l1, [l1])
	sublists(l2, l3.. ln)
}
all_sublists(l1, l2.. ln) = [sublists(l1, l2.. ln)]
