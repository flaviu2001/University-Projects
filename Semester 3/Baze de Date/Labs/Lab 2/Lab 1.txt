create table Player
(
    pid int primary key,
    name varchar(max),
    fide_rating int,
    country varchar(max),
    is_teacher bit
)

create table Game
(
    gid int primary key,
    white_pid int references Player(pid),
    black_pid int references Player(pid),
    game_date date,
	moves varchar(max),
	winner_pid int references Player(pid)
)

create table MemorableGames --one to many
(
    pid int references Player(pid),
    gid int references Game(gid),
    primary key (gid)
)

create table Chessboard
(
	cbid int primary key,
	board_size int,
	price int
)

create table GameChessboard --one to many
(
    gid int references Game(gid),
    cbid int references Chessboard(cbid),
    primary key (gid)
)

create table Tournament
(
    tid int primary key,
    date_begin date,
    date_end date
)

create table TournamentGame --one to many
(
	tid int references Tournament(tid),
	gid int references Game(gid),
	primary key (gid)
)

create table ChessClubs
(
    ccid int primary key,
    name varchar(max),
    country varchar(max),
    city varchar(max)
)

create table ChessClubsPlayers --many to many
(
    pid int references Player(pid),
    ccid int references ChessClubs(ccid),
    primary key (pid, ccid)
)
